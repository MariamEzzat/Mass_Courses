function openCommentsTabInSubTasks() {
    let commentsButtons = document.querySelectorAll('[data-TaskCommentID*="subT_comment_"]')
    commentsButtons.forEach(btnCommentView => {
        btnCommentView.addEventListener('click', function(e) {

            let subTaskid = e.target.attributes['data-SubtTaskID'].value
            let TaskID = e.target.attributes['data-TaskID'].value
            let divID = e.target.attributes['data-divID'].value

            fillCommentsListInSubTaskView(TaskID, subTaskid, `${divID}`)
        })
    })
};

/**
 * function that delete subtask after get object of subtask name and task id to send this object to removeSubtask function
 * @function deleteSubTaskS
 */
function deleteSubTaskS() {
    //when click add will get subttask name, taskID and send it to removeSubTask function
    let deleteSubTaskInViewSpecificTask = document.querySelectorAll('#taskList_specificTask #accordionSubtasks .deleteSubtask')
    deleteSubTaskInViewSpecificTask.forEach(deleteSubTask => {
        let removedSubTask = {};

        deleteSubTask.onclick = function(e) {
            //stop clickable row to click when click delete subTask
            document.querySelector('#warningAlertModal_close').innerHTML = "cancel"
            warningAlert(`Are Sure you want to delete '${e.target.title.trim()}' ?`, () => {
                document.querySelector('#warningAlertModal_close').innerHTML = "close"

                let subTaskName = e.target.attributes['data-title'].value.trim()
                e.stopPropagation();

                let IDofTask = specificTaskModal.querySelector('#taskList_specificTask_taskID').value
                removedSubTask = {
                    taskID: IDofTask,
                    name: subTaskName.trim()
                };

                removeSubTask(removedSubTask);



            }, () => {
                document.querySelector('#warningAlertModal_close').innerHTML = "close"
            })
        };
    });
};

function submitSubTaskComment(TaskID) {
    let subTaskCommentSubmit = document.querySelectorAll('#taskList_specificTask_subTaskComment_submit')
    subTaskCommentSubmit.forEach(btn => {
        btn.onclick = (e) => {
            e.target.classList.remove('button_Comment')
            e.target.classList.add('button_Loading')
            let commentValue = e.target.parentElement.previousElementSibling.value
            let subTaskid = e.target.attributes['data-SubtTaskID'].value
            let divID = e.target.attributes['data-divID'].value
            let obj = {
                taskID: TaskID,
                comment: commentValue,
                subtaskID: subTaskid
            }
            runLoadingAnimation('Loading ... .')
            post(
                IDENTIFIERS.TASKS.SUB_TASKS.NEW_COMMENT,
                obj,
                function(result) {
                    stopLoadingAnimation();
                    fillCommentsListInSubTaskView(TaskID, subTaskid, divID)
                    e.target.parentElement.previousElementSibling.value = ""
                },
                function(data) {
                    stopLoadingAnimation();
                }
            );
        };
    });
};
