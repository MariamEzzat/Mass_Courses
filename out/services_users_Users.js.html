/**

 * @function reciever - serv/Users.js
 * @param {*} envelope 
 * @param {*} attachedData 
 * @param {*} next 
 */
exports.reciever = (envelope, attachedData, next) => {
    try {
        functionMapper.get(envelope.serviceID)(
            envelope.data,
            attachedData,
            (respond) => {
                if (respond.status != "SUCCESS") {
                    /*AU
                    				// here to make it possible for future affected users on function.
                    				in case it's decided that affected users deleted in failed
                    				if (respond.hasOwnProperty("affectedUsers")) {
                    					affectedUsers.push(...respond.affectedUsers);
                    				}
                    */

                }
                let finalEnvelope = {
                    serviceID: envelope.serviceID,
                    status: respond.status,
                    data: respond.data
                }
                next(finalEnvelope);
            }
        );
    } catch (err) {
        console.log(SCRIPT_NAME + " error catched in reciever: " + JSON.stringify(err));
        next({
            serviceID: envelope.serviceID,
            status: "FAILED",
            data: err
        });
    }
}
